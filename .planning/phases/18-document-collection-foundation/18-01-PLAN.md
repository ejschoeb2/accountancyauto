  <files>supabase/migrations/20260224000001_document_collection_tables.sql</files>
  <action>
    Create the migration file with all five tables in this exact order (respects FK dependencies):

    **1. document_types** (global reference table — no org_id):
    - id UUID PK DEFAULT gen_random_uuid()
    - code TEXT NOT NULL UNIQUE (machine code: 'P60', 'BANK_STATEMENT', etc.)
    - label TEXT NOT NULL (display name)
    - description TEXT NOT NULL (portal checklist text)
    - default_retention_years INT NOT NULL
    - expected_mime_types TEXT[] (nullable)
    - created_at TIMESTAMPTZ NOT NULL DEFAULT now()
    - RLS: ENABLE ROW LEVEL SECURITY
    - Policy "document_types_select_authenticated": FOR SELECT TO authenticated USING(true)
    - Policy "document_types_service_role_all": FOR ALL TO service_role USING(true) WITH CHECK(true)

    **2. filing_document_requirements** (global reference table — no org_id):
    - id UUID PK DEFAULT gen_random_uuid()
    - filing_type_id TEXT NOT NULL REFERENCES filing_types(id)
    - document_type_id UUID NOT NULL REFERENCES document_types(id)
    - is_mandatory BOOLEAN NOT NULL DEFAULT false
    - created_at TIMESTAMPTZ NOT NULL DEFAULT now()
    - UNIQUE(filing_type_id, document_type_id)
    - RLS: ENABLE ROW LEVEL SECURITY
    - Policy "filing_document_requirements_select_authenticated": FOR SELECT TO authenticated USING(true)
    - Policy "filing_document_requirements_service_role_all": FOR ALL TO service_role USING(true) WITH CHECK(true)

    **3. client_documents** (org-scoped tenant table):
    - id UUID PK DEFAULT gen_random_uuid()
    - org_id UUID NOT NULL REFERENCES organisations(id)
    - client_id UUID NOT NULL REFERENCES clients(id) ON DELETE CASCADE
    - filing_type_id TEXT NOT NULL REFERENCES filing_types(id)
    - document_type_id UUID REFERENCES document_types(id) (nullable for unclassified)
    - storage_path TEXT NOT NULL
    - original_filename TEXT NOT NULL
    - received_at TIMESTAMPTZ NOT NULL DEFAULT now()
    - tax_period_end_date DATE NOT NULL (retention anchor — NOT NULL is critical)
    - retain_until DATE NOT NULL (derived at upload time)
    - retention_hold BOOLEAN NOT NULL DEFAULT false
    - retention_flagged BOOLEAN NOT NULL DEFAULT false
    - classification_confidence TEXT NOT NULL DEFAULT 'unclassified' CHECK(classification_confidence IN ('high', 'medium', 'low', 'unclassified'))
    - source TEXT NOT NULL CHECK(source IN ('inbound_email', 'portal_upload', 'manual'))
    - uploader_user_id UUID (nullable — null for inbound_email)
    - created_at TIMESTAMPTZ NOT NULL DEFAULT now()
    - RLS: ENABLE ROW LEVEL SECURITY
    - Policy "client_documents_select_org": FOR SELECT TO authenticated USING(org_id = auth_org_id())
    - Policy "client_documents_insert_org": FOR INSERT TO authenticated WITH CHECK(org_id = auth_org_id())
    - Policy "client_documents_update_org": FOR UPDATE TO authenticated USING(org_id = auth_org_id())
    - Policy "client_documents_delete_org": FOR DELETE TO authenticated USING(org_id = auth_org_id())
    - Policy "client_documents_service_role_all": FOR ALL TO service_role USING(true) WITH CHECK(true)
    - Indexes: idx_client_documents_org_id(org_id), idx_client_documents_client_id(client_id), idx_client_documents_filing_type(filing_type_id), idx_client_documents_retain_until(retain_until) WHERE NOT retention_hold (partial index for cron efficiency)

    **4. document_access_log** (org-scoped, INSERT-only for authenticated):
    - id UUID PK DEFAULT gen_random_uuid()
    - org_id UUID NOT NULL REFERENCES organisations(id)
    - document_id UUID NOT NULL REFERENCES client_documents(id) ON DELETE CASCADE
    - user_id UUID (nullable)
    - action TEXT NOT NULL CHECK(action IN ('view', 'download', 'delete'))
    - session_context TEXT (nullable)
    - created_at TIMESTAMPTZ NOT NULL DEFAULT now()
    - RLS: ENABLE ROW LEVEL SECURITY
    - Policy "document_access_log_insert_org": FOR INSERT TO authenticated WITH CHECK(org_id = auth_org_id())
    - Policy "document_access_log_select_org": FOR SELECT TO authenticated USING(org_id = auth_org_id())
    - NO UPDATE or DELETE policies for authenticated role (audit trail integrity)
    - Policy "document_access_log_service_role_all": FOR ALL TO service_role USING(true) WITH CHECK(true)
    - Indexes: idx_document_access_log_org_id(org_id), idx_document_access_log_document_id(document_id), idx_document_access_log_created_at(created_at DESC)

    **5. upload_portal_tokens** (org-scoped):
    - id UUID PK DEFAULT gen_random_uuid()
    - org_id UUID NOT NULL REFERENCES organisations(id)
    - client_id UUID NOT NULL REFERENCES clients(id) ON DELETE CASCADE
    - filing_type_id TEXT NOT NULL REFERENCES filing_types(id)
    - tax_year TEXT NOT NULL (e.g. '2024-25' for SA100, '2025' for company filings)
    - token_hash TEXT NOT NULL UNIQUE (SHA-256 hex of raw token — raw token NEVER stored)
    - expires_at TIMESTAMPTZ NOT NULL
    - used_at TIMESTAMPTZ (nullable — updated on each use)
    - created_at TIMESTAMPTZ NOT NULL DEFAULT now()
    - created_by_user_id UUID (nullable)
    - RLS: ENABLE ROW LEVEL SECURITY
    - Policy "upload_portal_tokens_select_org": FOR SELECT TO authenticated USING(org_id = auth_org_id())
    - Policy "upload_portal_tokens_insert_org": FOR INSERT TO authenticated WITH CHECK(org_id = auth_org_id())
    - Policy "upload_portal_tokens_update_org": FOR UPDATE TO authenticated USING(org_id = auth_org_id())
    - Policy "upload_portal_tokens_delete_org": FOR DELETE TO authenticated USING(org_id = auth_org_id())
    - Policy "upload_portal_tokens_service_role_all": FOR ALL TO service_role USING(true) WITH CHECK(true)
    - Indexes: idx_upload_portal_tokens_org_id(org_id), idx_upload_portal_tokens_token_hash(token_hash), idx_upload_portal_tokens_client_filing(client_id, filing_type_id, tax_year)
