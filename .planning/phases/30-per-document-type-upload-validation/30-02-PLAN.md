**1. Create `app/portal/[token]/components/validation-warning-card.tsx`**

New client component. Pattern mirrors the existing `ExtractionConfirmationCard` (green) but uses amber styling matching the duplicate warning pattern in `checklist-item.tsx` lines 108-134.

```typescript
'use client';

import { AlertTriangle } from 'lucide-react';

interface ValidationWarning {
  code: string;
  message: string;
  expected?: string;
  found?: string;
}

interface ValidationWarningCardProps {
  warnings: ValidationWarning[];
}

export function ValidationWarningCard({ warnings }: ValidationWarningCardProps) {
  return (
    <div className="mt-3 flex items-start gap-3 p-4 bg-amber-500/10 rounded-xl">
      <AlertTriangle className="size-5 text-amber-600 shrink-0 mt-0.5" />
      <div className="space-y-2">
        <p className="text-sm font-medium text-amber-600">
          {warnings.length === 1 ? 'Possible issue detected' : `${warnings.length} possible issues detected`}
        </p>
        <div className="space-y-1.5">
          {warnings.map((w, i) => (
            <p key={i} className="text-xs text-amber-600/80">{w.message}</p>
          ))}
        </div>
      </div>
    </div>
  );
}
```

Styling rules:
- `bg-amber-500/10` background (same as duplicate warning)
- `AlertTriangle` icon from lucide-react (same as duplicate warning)
- Amber text colors: `text-amber-600` for heading, `text-amber-600/80` for messages
- No Re-upload button — per CONTEXT.md, existing Replace button on checklist item is sufficient
