  <action>
    Create `app/api/cron/storage-health-check/route.ts`. Follow the exact structure of `app/api/cron/trial-reminder/route.ts`.

    ```typescript
    import { NextRequest, NextResponse } from "next/server";
    import { createAdminClient } from "@/lib/supabase/admin";
    import { decryptToken } from "@/lib/crypto/tokens";
    import { ServerClient } from "postmark";

    export const dynamic = "force-dynamic";
    export const maxDuration = 300;

    export async function GET(request: NextRequest) {
      // 1. Auth check
      const authHeader = request.headers.get("authorization");
      if (authHeader !== `Bearer ${process.env.CRON_SECRET}`) {
        return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
      }

      const admin = createAdminClient();

      // 2. Fetch orgs with active non-Supabase backends; skip reauth_required (stronger signal)
      const { data: orgs, error: fetchError } = await admin
        .from("organisations")
        .select("id, name, storage_backend, storage_backend_status, google_access_token_enc, google_refresh_token_enc, ms_token_cache_enc, ms_home_account_id, dropbox_access_token_enc, dropbox_refresh_token_enc, dropbox_token_expires_at")
        .neq("storage_backend", "supabase")
        .neq("storage_backend_status", "reauth_required")  // skip: already flagged for admin
        .in("subscription_status", ["active", "trialing"]);

      if (fetchError) {
        return NextResponse.json({ error: fetchError.message }, { status: 500 });
      }

      if (!orgs || orgs.length === 0) {
        return NextResponse.json({ success: true, checked: 0, errors: 0, skipped: 0 });
      }

      let checked = 0;
      let errors = 0;
      let skipped = 0;

      for (const org of orgs) {
        let healthy = false;

        try {
          if (org.storage_backend === "google_drive") {
            healthy = await checkGoogleDriveHealth(org);
          } else if (org.storage_backend === "onedrive") {
            healthy = await checkOneDriveHealth(org, admin);
          } else if (org.storage_backend === "dropbox") {
            healthy = await checkDropboxHealth(org);
          } else {
            skipped++;
            continue;
          }
        } catch {
          healthy = false;
        }

        if (healthy) {
          // Recovery: clear error status and reset idempotency flag
          if (org.storage_backend_status === "error") {
            await admin.from("organisations").update({ storage_backend_status: "active" }).eq("id", org.id);
            await admin.from("app_settings").delete().eq("org_id", org.id).eq("key", "storage_health_error_notified");
            console.log(`[Cron:storage-health-check] Org ${org.id} recovered — status set to active`);
          }
        } else {
          errors++;
          // Set status to 'error'
          await admin.from("organisations").update({ storage_backend_status: "error" }).eq("id", org.id);

          // Idempotency: only email once per error period
          const { data: alreadyNotified } = await admin
            .from("app_settings")
            .select("id")
            .eq("org_id", org.id)
            .eq("key", "storage_health_error_notified")
            .eq("value", "true")
            .single();

          if (!alreadyNotified) {
            // Fetch org admin email
            const { data: adminMemberships } = await admin
              .from("user_organisations")
              .select("user_id")
              .eq("org_id", org.id)
              .eq("role", "admin")
              .limit(1);

            const adminUserId = adminMemberships?.[0]?.user_id;
            if (adminUserId) {
              try {
                const { data: { user: adminUser } } = await admin.auth.admin.getUserById(adminUserId);
                if (adminUser?.email) {
                  const mailer = new ServerClient(process.env.POSTMARK_SERVER_TOKEN!);
                  await mailer.sendEmail({
                    From: `Prompt <noreply@${process.env.POSTMARK_SENDER_DOMAIN ?? "phasetwo.uk"}>`,
                    To: adminUser.email,
                    Subject: `Action required: ${org.storage_backend === "google_drive" ? "Google Drive" : org.storage_backend === "onedrive" ? "Microsoft OneDrive" : "Dropbox"} storage connection issue`,
                    TextBody: `Hi,\n\nYour ${org.storage_backend === "google_drive" ? "Google Drive" : org.storage_backend === "onedrive" ? "Microsoft OneDrive" : "Dropbox"} storage connection for ${org.name} on Prompt is experiencing issues. New documents may not be uploaded to your provider.\n\nPlease visit Settings > Storage to check your connection and reconnect if needed.\n\nhttps://app.phasetwo.uk/settings?tab=storage\n\nThe Prompt Team`,
                  });
                  // Mark as notified
                  await admin.from("app_settings").upsert(
                    { org_id: org.id, key: "storage_health_error_notified", value: "true" },
                    { onConflict: "org_id,key" }
                  );
                }
              } catch (emailErr) {
                console.error(`[Cron:storage-health-check] Failed to send notification for org ${org.id}:`, emailErr);
              }
            }
          }
        }

        checked++;
      }

      return NextResponse.json({ success: true, checked, errors, skipped });
    }

    async function checkGoogleDriveHealth(org: { google_access_token_enc: string | null; google_refresh_token_enc: string | null }): Promise<boolean> {
      if (!org.google_access_token_enc) return false;
      const { auth: googleAuth } = await import("@googleapis/drive");
      const accessToken = await decryptToken(org.google_access_token_enc);
      const oauth2Client = new googleAuth.OAuth2(
        process.env.GOOGLE_CLIENT_ID,
        process.env.GOOGLE_CLIENT_SECRET
      );
      oauth2Client.setCredentials({ access_token: accessToken });
      const { drive: createDrive } = await import("@googleapis/drive");
      const drive = createDrive({ version: "v3", auth: oauth2Client });
      const res = await drive.about.get({ fields: "user" });
      return res.status === 200;
    }

    async function checkOneDriveHealth(org: { ms_token_cache_enc: string | null; ms_home_account_id: string | null; id: string }, admin: ReturnType<typeof createAdminClient>): Promise<boolean> {
      if (!org.ms_token_cache_enc || !org.ms_home_account_id) return false;
      // Use the existing PostgresMsalCachePlugin pattern from lib/storage/onedrive.ts
      // to acquire a token, then make a lightweight Graph call
      const { ConfidentialClientApplication } = await import("@azure/msal-node");
      const { PostgresMsalCachePlugin } = await import("@/lib/storage/postgres-msal-cache-plugin");
      const cachePlugin = new PostgresMsalCachePlugin(admin, org.id);
      const msalClient = new ConfidentialClientApplication({
        auth: {
          clientId: process.env.AZURE_CLIENT_ID!,
          clientSecret: process.env.AZURE_CLIENT_SECRET!,
          authority: `https://login.microsoftonline.com/common`,
        },
        cache: { cachePlugin },
      });
      const tokenResponse = await msalClient.acquireTokenSilent({
        scopes: ["Files.ReadWrite", "offline_access"],
        account: { homeAccountId: org.ms_home_account_id } as any,
      });
      if (!tokenResponse?.accessToken) return false;
      // Make real Graph call to confirm end-to-end access
      const resp = await fetch("https://graph.microsoft.com/v1.0/me/drive", {
        headers: { Authorization: `Bearer ${tokenResponse.accessToken}` },
      });
      return resp.ok;
    }

    async function checkDropboxHealth(org: { dropbox_access_token_enc: string | null; dropbox_refresh_token_enc: string | null; dropbox_token_expires_at: string | null }): Promise<boolean> {
      if (!org.dropbox_refresh_token_enc) return false;
      const { DropboxAuth, Dropbox } = await import("dropbox");
      const refreshToken = await decryptToken(org.dropbox_refresh_token_enc);
      const dbxAuth = new DropboxAuth({
        clientId: process.env.DROPBOX_APP_KEY,
        clientSecret: process.env.DROPBOX_APP_SECRET,
        refreshToken,
      });
      await dbxAuth.checkAndRefreshAccessToken();
      const dbx = new Dropbox({ auth: dbxAuth });
      const result = await dbx.usersGetCurrentAccount();
      return !!result?.result?.account_id;
    }
    ```

    Important implementation notes:
    - Use dynamic `import()` for provider SDKs to keep cold start fast; providers are only imported on demand per org
    - The `neq("storage_backend_status", "reauth_required")` filter prevents the cron from overwriting the stronger `reauth_required` signal with `error`
    - The `user_id IS NULL` constraint is NOT used in app_settings for the idempotency key — use `.eq("org_id", org.id).eq("key", "storage_health_error_notified")` without user_id filter (org-level setting). However, the app_settings upsert must include `user_id: null` in the payload because the table has a unique constraint on `(org_id, user_id, key)`. Update the upsert payload to: `{ org_id: org.id, user_id: null, key: "storage_health_error_notified", value: "true" }` and `onConflict: "org_id,user_id,key"`.
    - If `PostgresMsalCachePlugin` does not exist as a named import at `@/lib/storage/postgres-msal-cache-plugin`, look for it in `lib/storage/onedrive.ts` or `lib/storage/msal-cache-plugin.ts` — check the actual file path before importing. Use the same import path used in the OneDrive provider file.
    - For Google Drive: if `google_refresh_token_enc` is available but access token is expired, the OAuth2 client will attempt to refresh automatically when making the API call. This is acceptable for the health-check.
    - `maxDuration = 300` is set because the cron may iterate many orgs with network calls.
