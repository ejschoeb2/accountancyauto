**1. Add server actions to `app/actions/settings.ts`:**

Add two new exported functions:

`getPostmarkSettings()`: Query the `organisations` table for the current org's `postmark_server_token` and `postmark_sender_domain`. Use `getOrgId()` to get org_id, then query with the admin client (`createAdminClient()`) since the organisations table has restricted RLS:
```typescript
export async function getPostmarkSettings(): Promise<{ token: string; senderDomain: string }> {
  const orgId = await getOrgId();
  const admin = createAdminClient();
  const { data: org } = await admin
    .from('organisations')
    .select('postmark_server_token, postmark_sender_domain')
    .eq('id', orgId)
    .single();
  return {
    token: org?.postmark_server_token || '',
    senderDomain: org?.postmark_sender_domain || '',
  };
}
```
Import `createAdminClient` from `@/lib/supabase/admin` (add to existing imports).

`updatePostmarkSettings(token: string, senderDomain: string)`: Validate inputs, then update the `organisations` table. Use admin client since organisations table is service_role-writable only:
```typescript
export async function updatePostmarkSettings(token: string, senderDomain: string): Promise<{ error?: string }> {
  const orgId = await getOrgId();
  await requireWriteAccess(orgId);
  const admin = createAdminClient();
  const { error } = await admin
    .from('organisations')
    .update({ postmark_server_token: token.trim() || null, postmark_sender_domain: senderDomain.trim() || null })
    .eq('id', orgId);
  if (error) return { error: error.message };
  return {};
}
```
