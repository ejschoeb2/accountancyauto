  documentTypeId: string | null;

Logic:
1. Build a static KEYWORD_MAP array covering ALL 23 seeded document_types codes. Each entry has `patterns: RegExp[]`, `code: string`, `filingType: string | null`. Cover at minimum:
   - P60: [/\bp60\b/i, /end.of.year.cert/i], filingType: 'self_assessment'
   - P45: [/\bp45\b/i, /leaving.cert/i], filingType: 'self_assessment'
   - P11D: [/\bp11d\b/i, /expenses.and.benefits/i, /benefit.in.kind/i], filingType: 'self_assessment'
   - SA302: [/\bsa302\b/i, /tax.calculation/i, /hmrc.tax.calc/i], filingType: 'self_assessment'
   - BANK_STATEMENT: [/bank.statement/i, /\bstatement\b/i], filingType: null
   - DIVIDEND_VOUCHER: [/dividend.voucher/i, /div.voucher/i], filingType: null
   - VAT_RETURN_WORKINGS: [/vat.return/i, /vat.workings/i], filingType: 'vat_return'
   - PURCHASE_INVOICES: [/purchase.invoice/i, /supplier.invoice/i], filingType: 'vat_return'
   - SALES_INVOICES: [/sales.invoice/i, /customer.invoice/i], filingType: 'vat_return'
   - RECEIPT: [/\breceipt\b/i], filingType: null
   - PAYSLIP: [/payslip/i, /pay.slip/i], filingType: 'self_assessment'
   - COMPANY_ACCOUNTS: [/company.accounts/i, /annual.accounts/i, /statutory.accounts/i], filingType: 'companies_house'
   - CONFIRMATION_STATEMENT: [/confirmation.statement/i], filingType: 'companies_house'
   - CT600: [/\bct600\b/i, /corporation.tax.return/i], filingType: 'ct600_filing'
   - CORPORATION_TAX_PAYMENT: [/corporation.tax.payment/i, /ct.payment/i], filingType: 'ct600_filing'
   - Add remaining types from document_types seed data in migration 20260224000001

2. Classification algorithm:
   - Step 1: Iterate KEYWORD_MAP; if any pattern matches the filename (case-insensitive):
     - Query document_types by code to get the UUID id
     - Also check if mimeType matches expected_mime_types from document_types (query DB)
     - Return confidence 'high' if MIME also matches, 'medium' if only filename matches
   - Step 2: If no keyword match but mimeType is 'application/pdf': return confidence 'low', all IDs null
   - Step 3: Otherwise return confidence 'unclassified', all IDs null

3. For the document_types DB query (to get ID from code), use a single query at the start to cache all codes:

Add the `processAttachments` function as a module-level async function in the same file:

```typescript
async function processAttachments(
  attachments: PostmarkInboundWebhook['Attachments'],
  inboundEmailId: string,
  client: { id: string; org_id: string; year_end_date?: string | null } | null,
  orgId: string,
  supabase: ReturnType<typeof createServiceClient>
): Promise<void> {
  const { uploadDocument } = await import('@/lib/documents/storage');
  const { classifyDocument } = await import('@/lib/documents/classify');
  const { calculateRetainUntil } = await import('@/lib/documents/metadata');

  for (const attachment of attachments) {
    try {
      const fileBuffer = Buffer.from(attachment.Content, 'base64');
      const classification = await classifyDocument(attachment.Name, attachment.ContentType, supabase);

      // Derive tax period end from client year_end_date (best-effort; accountant can correct)
      // Use the most recently completed year end before today
      const now = new Date();
      let taxPeriodEndDate: Date;
      if (client?.year_end_date) {
        const yearEnd = new Date(client.year_end_date);
        // Use this year's year-end if it has passed, otherwise previous year
        taxPeriodEndDate = yearEnd <= now
          ? yearEnd
          : new Date(yearEnd.getFullYear() - 1, yearEnd.getMonth(), yearEnd.getDate());
      } else {
        // Fallback: use previous 5 April (UK tax year end)
        const apr5 = new Date(now.getFullYear(), 3, 5);
        taxPeriodEndDate = apr5 <= now ? apr5 : new Date(now.getFullYear() - 1, 3, 5);
      }

      const taxYear = `${taxPeriodEndDate.getFullYear()}`;
      const filingTypeId = classification.filingTypeId ?? 'ct600_filing';

      // For unmatched emails (no client), skip storage — cannot scope the path
      if (!client?.id) {
        console.warn('[Postmark Inbound] No client match — skipping attachment storage for:', attachment.Name);
        continue;
      }

      const { storagePath } = await uploadDocument({
        orgId,
        clientId: client.id,
        filingTypeId,
        taxYear,
        file: fileBuffer,
        originalFilename: attachment.Name,
        mimeType: attachment.ContentType,
      });

      const retainUntil = calculateRetainUntil(filingTypeId as FilingTypeId, taxPeriodEndDate);

      await supabase.from('client_documents').insert({
        org_id: orgId,
        client_id: client.id,
        filing_type_id: filingTypeId,
        document_type_id: classification.documentTypeId,
        storage_path: storagePath,
        original_filename: attachment.Name,
        tax_period_end_date: taxPeriodEndDate.toISOString().split('T')[0],
        retain_until: retainUntil.toISOString().split('T')[0],
        classification_confidence: classification.confidence,
        source: 'inbound_email',
        inbound_email_id: inboundEmailId,
      });

      console.log('[Postmark Inbound] Attachment stored:', {
        filename: attachment.Name,
        confidence: classification.confidence,
        code: classification.documentTypeCode,
      });
    } catch (err) {
      console.error('[Postmark Inbound] Failed to process attachment:', attachment.Name, err);
      // Continue to next attachment — never throw from this loop
    }
  }
}
```
